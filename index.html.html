// ═══════════════════════════════════════════ // OTC Online Room — Google Sheets Sync Script // ═══════════════════════════════════════════ var NAVY = "#1A2B4A"; var ACCENT = "#2563EB"; var WHITE = "#FFFFFF"; var LIGHT_BLUE = "#EFF6FF"; var HEADER_BG = "#DBEAFE"; var ROW_ALT = "#F8FAFC"; var BORDER = "#CBD5E1"; var GREEN = "#16A34A"; var GREEN_BG = "#F0FDF4"; var AMBER = "#D97706"; var AMBER_BG = "#FFFBEB"; var RED = "#DC2626"; var RED_BG = "#FEF2F2"; var GRAY = "#475569"; var TOTAL_BG = "#EFF6FF"; function doPost(e) { try { var raw = (e.postData && e.postData.contents) ? e.postData.contents : (e.parameter && e.parameter.payload ? e.parameter.payload : null); if (!raw) return jsonOut({ status: "error", message: "No data received" }); var data = JSON.parse(raw); if (data.action === "sync_all") return syncAll(data.products); return jsonOut({ status: "unknown_action" }); } catch(err) { return jsonOut({ status: "error", message: err.message }); } } function doGet(e) { if (e && e.parameter && e.parameter.payload) { try { var data = JSON.parse(e.parameter.payload); if (data.action === "sync_all") return syncAll(data.products); } catch(err) { return jsonOut({ status: "error", message: err.message }); } } return jsonOut({ status: "ok", message: "OTC Sync endpoint active" }); } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function syncAll(products) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Remove old sheet if exists, create fresh ── var old = ss.getSheetByName("OTC Products"); if (old) ss.deleteSheet(old); var sh = ss.insertSheet("OTC Products"); ss.setActiveSheet(sh); var now = new Date(); var dateStr = Utilities.formatDate(now, ss.getSpreadsheetTimeZone(), "dd MMM yyyy, HH:mm"); var total = products.length; var inStock = products.filter(function(p){ return p.qty > 0; }).length; var outStock = products.filter(function(p){ return p.qty === 0; }).length; var units = products.reduce(function(a,p){ return a + p.qty; }, 0); // ════════════════════════════════ // ROW 1 — Main Title // ════════════════════════════════ sh.getRange("A1:E1").merge() .setValue("OTC Online Room — Product Inventory") .setFontFamily("Arial").setFontSize(18).setFontWeight("bold") .setFontColor(NAVY).setBackground(WHITE) .setVerticalAlignment("middle").setHorizontalAlignment("left"); sh.setRowHeight(1, 42); // ROW 2 — Subtitle / date sh.getRange("A2:E2").merge() .setValue("Generated: " + dateStr + " | Total Products: " + total) .setFontFamily("Arial").setFontSize(10).setFontColor(GRAY) .setBackground(WHITE).setHorizontalAlignment("left") .setVerticalAlignment("middle"); sh.setRowHeight(2, 22); // ROW 3 — thin accent line (simulated with bg) sh.getRange("A3:E3").merge().setBackground(ACCENT); sh.setRowHeight(3, 3); // ROW 4 — spacer sh.getRange("A4:E4").merge().setBackground(WHITE); sh.setRowHeight(4, 8); // ════════════════════════════════ // ROWS 5-6 — Summary Cards // ════════════════════════════════ var cards = [ { col: "A", label: "Total Products", value: total, bg: "#EFF6FF", fg: "#2563EB", border: "#BFDBFE" }, { col: "B", label: "In Stock", value: inStock, bg: "#F0FDF4", fg: "#16A34A", border: "#BBF7D0" }, { col: "C", label: "Out of Stock", value: outStock, bg: "#FEF2F2", fg: "#DC2626", border: "#FECACA" }, { col: "D", label: "Total Units", value: units, bg: "#EFF6FF", fg: "#1A2B4A", border: "#BFDBFE" }, ]; cards.forEach(function(c) { sh.getRange(c.col + "5").setValue(c.label) .setFontFamily("Arial").setFontSize(9).setFontColor(GRAY) .setBackground(c.bg).setHorizontalAlignment("center") .setVerticalAlignment("bottom").setFontWeight("normal"); sh.getRange(c.col + "6").setValue(c.value) .setFontFamily("Arial").setFontSize(20).setFontWeight("bold") .setFontColor(c.fg).setBackground(c.bg) .setHorizontalAlignment("center").setVerticalAlignment("top"); }); sh.setRowHeight(5, 20); sh.setRowHeight(6, 30); // ROW 7 — spacer sh.getRange("A7:E7").merge().setBackground(WHITE); sh.setRowHeight(7, 10); // ════════════════════════════════ // ROW 8 — Table Header // ════════════════════════════════ var headers = ["SKU", "Product Name", "Quantity", "Status", "Last Updated"]; var hRow = sh.getRange("A8:E8"); hRow.setValues([headers]) .setFontFamily("Arial").setFontSize(10).setFontWeight("bold") .setFontColor(WHITE).setBackground(NAVY) .setHorizontalAlignment("center").setVerticalAlignment("middle"); sh.getRange("B8").setHorizontalAlignment("left"); sh.setRowHeight(8, 26); // ════════════════════════════════ // DATA ROWS — starting row 9 // ════════════════════════════════ var syncTime = Utilities.formatDate(now, ss.getSpreadsheetTimeZone(), "dd MMM yyyy HH:mm"); for (var i = 0; i < products.length; i++) { var p = products[i]; var row = 9 + i; var qty = p.qty; var status = qty > 1 ? "In Stock" : (qty === 1 ? "Low Stock" : "Out of Stock"); var qtyColor = qty > 1 ? GREEN : (qty === 1 ? AMBER : RED); var qtyBg = qty > 1 ? GREEN_BG : (qty === 1 ? AMBER_BG : RED_BG); var statusColor = qtyColor; var statusBg = qtyBg; var rowBg = (i % 2 === 0) ? WHITE : ROW_ALT; // SKU sh.getRange(row, 1).setValue(p.sku) .setFontFamily("Arial").setFontSize(9).setFontColor(GRAY) .setBackground(rowBg).setHorizontalAlignment("center") .setVerticalAlignment("middle"); // Product Name sh.getRange(row, 2).setValue(p.name) .setFontFamily("Arial").setFontSize(9.5).setFontColor("#1E293B") .setBackground(rowBg).setHorizontalAlignment("left") .setVerticalAlignment("middle"); // Quantity sh.getRange(row, 3).setValue(qty) .setFontFamily("Arial").setFontSize(11).setFontWeight("bold") .setFontColor(qtyColor).setBackground(qtyBg) .setHorizontalAlignment("center").setVerticalAlignment("middle"); // Status sh.getRange(row, 4).setValue(status) .setFontFamily("Arial").setFontSize(9).setFontWeight("bold") .setFontColor(statusColor).setBackground(statusBg) .setHorizontalAlignment("center").setVerticalAlignment("middle"); // Last Updated sh.getRange(row, 5).setValue(syncTime) .setFontFamily("Arial").setFontSize(9).setFontColor(GRAY) .setBackground(rowBg).setHorizontalAlignment("center") .setVerticalAlignment("middle"); sh.setRowHeight(row, 20); } // ── Grid border on data range ── var dataRange = sh.getRange(8, 1, products.length + 1, 5); dataRange.setBorder(true, true, true, true, true, true, BORDER, SpreadsheetApp.BorderStyle.SOLID); // ════════════════════════════════ // TOTALS ROW // ════════════════════════════════ var totalRow = 9 + products.length + 1; sh.getRange(totalRow, 1, 1, 2).merge() .setValue("TOTAL UNITS IN STOCK") .setFontFamily("Arial").setFontSize(10).setFontWeight("bold") .setFontColor(NAVY).setBackground(LIGHT_BLUE) .setHorizontalAlignment("right").setVerticalAlignment("middle"); sh.getRange(totalRow, 3) .setFormula("=SUM(C9:C" + (totalRow - 2) + ")") .setFontFamily("Arial").setFontSize(12).setFontWeight("bold") .setFontColor(ACCENT).setBackground(LIGHT_BLUE) .setHorizontalAlignment("center").setVerticalAlignment("middle"); sh.getRange(totalRow, 4, 1, 2).merge().setBackground(LIGHT_BLUE); sh.setRowHeight(totalRow, 26); // ── Footer ── var footerRow = totalRow + 2; sh.getRange(footerRow, 1, 1, 5).merge() .setValue("OTC Online Room · Confidential Internal Document · " + dateStr) .setFontFamily("Arial").setFontSize(8).setFontColor("#94A3B8") .setHorizontalAlignment("center").setFontStyle("italic"); // ════════════════════════════════ // COLUMN WIDTHS // ════════════════════════════════ sh.setColumnWidth(1, 110); // SKU sh.setColumnWidth(2, 360); // Product Name sh.setColumnWidth(3, 80); // Qty sh.setColumnWidth(4, 110); // Status sh.setColumnWidth(5, 150); // Last Updated // ── Freeze header row ── sh.setFrozenRows(8); // ── Hide gridlines for clean look ── sh.setHiddenGridlines(true); // ── Row 1 height ── sh.setRowHeight(1, 46); return jsonOut({ status: "ok", synced: products.length }); }
